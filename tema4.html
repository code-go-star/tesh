4. Estructuras no lineales

4.1 Árboles

  • 4.1.1 Clasificación de árboles

    • Los árboles se clasifican según su estructura y propiedades, como árboles binarios, árboles binarios de búsqueda, árboles balanceados, y más.

  • 4.1.2 Operaciones básicas sobre árboles binarios

    • Incluyen inserción, eliminación, búsqueda, y recorrido (preorden, inorden, postorden).

  • 4.1.3 Aplicaciones

    • Se utilizan en bases de datos, sistemas de archivos, algoritmos de búsqueda y más.

4.2 Grafos

  • 4.2.1 Representación de grafos

    • Los grafos pueden representarse mediante matrices de adyacencia, listas de adyacencia o listas de incidencia.

  • 4.2.2 Operaciones básicas

    • Incluyen la búsqueda en profundidad (DFS), búsqueda en anchura (BFS), y encontrar caminos mínimos.

Ejemplo en Java


public class ArbolBinario {
    // Clase Nodo
    static class Nodo {
        int valor;
        Nodo izquierdo, derecho;

        public Nodo(int valor) {
            this.valor = valor;
            izquierdo = derecho = null;
        }
    }

    // Raiz del árbol binario
    Nodo raiz;

    // Método para insertar un nuevo nodo
    void insertar(int valor) {
        raiz = insertarRec(raiz, valor);
    }

    // Función recursiva para insertar un nuevo nodo
    Nodo insertarRec(Nodo raiz, int valor) {
        if (raiz == null) {
            raiz = new Nodo(valor);
            return raiz;
        }

        if (valor < raiz.valor)
            raiz.izquierdo = insertarRec(raiz.izquierdo, valor);
        else if (valor > raiz.valor)
            raiz.derecho = insertarRec(raiz.derecho, valor);

        return raiz;
    }

    // Recorrido en preorden
    void preOrden(Nodo raiz) {
        if (raiz != null) {
            System.out.print(raiz.valor + " ");
            preOrden(raiz.izquierdo);
            preOrden(raiz.derecho);
        }
    }

    // Recorrido en inorden
    void inOrden(Nodo raiz) {
        if (raiz != null) {
            inOrden(raiz.izquierdo);
            System.out.print(raiz.valor + " ");
            inOrden(raiz.derecho);
        }
    }

    // Recorrido en postorden
    void postOrden(Nodo raiz) {
        if (raiz != null) {
            postOrden(raiz.izquierdo);
            postOrden(raiz.derecho);
            System.out.print(raiz.valor + " ");
        }
    }

    public static void main(String[] args) {
        ArbolBinario arbol = new ArbolBinario();

        /* Insertar elementos */
        arbol.insertar(50);
        arbol.insertar(30);
        arbol.insertar(20);
        arbol.insertar(40);
        arbol.insertar(70);
        arbol.insertar(60);
        arbol.insertar(80);

        System.out.print("Recorrido en preorden: ");
        arbol.preOrden(arbol.raiz);
        System.out.println();

        System.out.print("Recorrido en inorden: ");
        arbol.inOrden(arbol.raiz);
        System.out.println();

        System.out.print("Recorrido en postorden: ");
        arbol.postOrden(arbol.raiz);
        System.out.println();
    }
}
