Temario

6 Métodos de búsqueda

6.1 Búsqueda secuencial

La búsqueda secuencial es un algoritmo simple en el que se verifica cada elemento de la lista uno por uno hasta que se encuentra el elemento deseado o se recorre toda la lista.

6.2 Búsqueda binaria

La búsqueda binaria es un algoritmo eficiente que se utiliza en listas ordenadas. Divide repetidamente el intervalo de búsqueda a la mitad hasta encontrar el elemento deseado.

6.3 Búsqueda por funciones de hash

La búsqueda por funciones de hash utiliza una función hash para mapear claves a ubicaciones específicas en una tabla hash, permitiendo un acceso rápido a los elementos.

Ejemplo en Java


import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;

public class MetodosDeBusqueda {

    // Búsqueda secuencial
    public static int busquedaSecuencial(int[] arr, int x) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == x) {
                return i;
            }
        }
        return -1;
    }

    // Búsqueda binaria
    public static int busquedaBinaria(int[] arr, int x) {
        int izquierda = 0, derecha = arr.length - 1;
        while (izquierda <= derecha) {
            int medio = izquierda + (derecha - izquierda) / 2;

            if (arr[medio] == x)
                return medio;
            
            if (arr[medio] < x)
                izquierda = medio + 1;
            else
                derecha = medio - 1;
        }
        return -1;
    }

    // Búsqueda por funciones de hash
    public static Integer busquedaHash(Map hashMap, int x) {
        return hashMap.get(x);
    }

    public static void main(String[] args) {
        int[] arr = {2, 3, 4, 10, 40};
        int x = 10;

        // Búsqueda secuencial
        int resultadoSecuencial = busquedaSecuencial(arr, x);
        System.out.println("Búsqueda secuencial: Elemento encontrado en el índice " + resultadoSecuencial);

        // Búsqueda binaria (requiere un arreglo ordenado)
        int resultadoBinaria = busquedaBinaria(arr, x);
        System.out.println("Búsqueda binaria: Elemento encontrado en el índice " + resultadoBinaria);

        // Búsqueda por funciones de hash
        Map hashMap = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            hashMap.put(arr[i], i);
        }
        Integer resultadoHash = busquedaHash(hashMap, x);
        System.out.println("Búsqueda por funciones de hash: Elemento encontrado en el índice " + resultadoHash);
    }
}
